កិច្ចសន្យាអភិវឌ្ឍន៍ប្រព័ន្ធកម្មវិធី

ប្រព័ន្ធគ្រប់គ្រងឥណទាន និងគណនេយយ្យ (Loan Installment & Accounting System)

កិច្ចសន្យាអភិវឌ្ឍន៍ប្រព័ន្ធកម្មវិធីនេះ ត្រូវបានធ្វើឡើងនៅថ្ងៃទី [ថ្ងៃ] ខែ [ខែ] ឆ្នាំ [ឆ្នាំ] រវាង:

ភាគីទី ១ (អតិថិជន):

ឈ្មោះក្រុមហ៊ុន: [ឈ្មោះក្រុមហ៊ុនអតិថិជន]

អាសយដ្ឋាន: [អាសយដ្ឋានក្រុមហ៊ុនអតិថិជន]

តំណាងដោយ: លោក/លោកស្រី [ឈ្មោះតំណាងអតិថិជន] (តួនាទី: [តួនាទីតំណាង])

(តទៅនេះហៅកាត់ថា "អតិថិជន")

និង

ភាគីទី ២ (អ្នកអភិវឌ្ឍន៍):

ឈ្មោះបុគ្គល/ក្រុមហ៊ុន: [ឈ្មោះអ្នក/ក្រុមហ៊ុនរបស់អ្នក]

អាសយដ្ឋាន: [អាសយដ្ឋានរបស់អ្នក/ក្រុមហ៊ុនរបស់អ្នក]

(តទៅនេះហៅកាត់ថា "អ្នកអភិវឌ្ឍន៍")

ដោយយល់ឃើញថា អតិថិជនមានបំណងចង់អភិវឌ្ឍន៍ប្រព័ន្ធគ្រប់គ្រងឥណទាន និងគណនេយ្យ ហើយអ្នកអភិវឌ្ឍន៍មានសមត្ថភាព និងបទពិសោធន៍ក្នុងការអនុវត្តការងារនេះ ភាគីទាំងពីរបានព្រមព្រៀងគ្នាលើលក្ខខណ្ឌដូចខាងក្រោម:

  1. I. វិសាលភាពនៃការងារ (Scope of Work)

    អ្នកអភិវឌ្ឍន៍នឹងរចនា អភិវឌ្ឍន៍ ធ្វើតេស្ត និងដាក់ឱ្យដំណើរការនូវប្រព័ន្ធគ្រប់គ្រងឥណទាន និងគណនេយ្យ ("ប្រព័ន្ធ") ដូចដែលបានរៀបរាប់លម្អិតនៅក្នុង **ឯកសារវិភាគតម្រូវការ (Requirement Document / Scope of Work)** ដែលជាផ្នែកមួយនៃកិច្ចសន្យានេះ។ មុខងារសំខាន់ៗរួមមាន:

    • ប្រព័ន្ធគ្រប់គ្រងឥណទាន (Loan Management Module)
    • ប្រព័ន្ធគណនេយ្យ (Accounting Module)
    • ការគ្រប់គ្រងអ្នកប្រើប្រាស់ និងសន្តិសុខ (User & Security Management)
    • ការបង្កើតរបាយការណ៍ (Reporting)
    • និងមុខងារផ្សេងៗទៀតដែលបានព្រមព្រៀងគ្នានៅក្នុងឯកសារវិភាគតម្រូវការ។

  2. II. កាលបរិច្ឆេទ និងផែនការការងារ (Timeline and Work Plan)

    ការងារអភិវឌ្ឍន៍នឹងត្រូវអនុវត្តតាមផែនការ និងកាលបរិច្ឆេទដូចដែលបានរៀបរាប់លម្អិតនៅក្នុង **ឯកសារសំណើគម្រោង (Project Proposal)** ដែលភ្ជាប់មកជាមួយកិច្ចសន្យានេះ។ កាលបរិច្ឆេទប៉ាន់ស្មានសម្រាប់ការបញ្ចប់គម្រោងគឺ [សរុបពេលវេលា] សប្ដាហ៍/ខែ ចាប់ពីថ្ងៃចុះហត្ថលេខាលើកិច្ចសន្យានេះ។

  3. III. តម្លៃ និងលក្ខខណ្ឌនៃការទូទាត់ (Payment Terms)

    តម្លៃសរុបសម្រាប់សេវាកម្មអភិវឌ្ឍន៍ប្រព័ន្ធនេះគឺ [X,XXX] ដុល្លារអាមេរិក ($X,XXX.00)។

    ការទូទាត់នឹងត្រូវធ្វើឡើងជាដំណាក់កាលដូចខាងក្រោម:

    • ប្រាក់កក់ ៣០% ($[Y,YYY]) ត្រូវបង់នៅពេលចុះហត្ថលេខាលើកិច្ចសន្យា។
    • ៣០% បន្ទាប់ ($[Y,YYY]) ត្រូវបង់នៅពេលបញ្ចប់ដំណាក់កាលទី ១ (ការវិភាគ និងរចនា)។
    • ៣០% បន្ទាប់ ($[Y,YYY]) ត្រូវបង់នៅពេលបញ្ចប់ដំណាក់កាលទី ២ (ការអភិវឌ្ឍន៍)។
    • ១០% ចុងក្រោយ ($[Y,YYY]) ត្រូវបង់នៅពេលដាក់ឱ្យដំណើរការប្រព័ន្ធ និងបណ្ដុះបណ្ដាលរួចរាល់។

    សម្គាល់: តម្លៃនេះមិនរាប់បញ្ចូលថ្លៃ Server Hosting, Domain Name, License software ភាគីទីបី ឬថ្លៃសេវាកម្មផ្សេងៗដែលមិនបានរៀបរាប់ក្នុងឯកសារវិសាលភាពការងារឡើយ។

  4. IV. ការទទួលខុសត្រូវរបស់ភាគី

    IV.១. ការទទួលខុសត្រូវរបស់អ្នកអភិវឌ្ឍន៍:

    • អភិវឌ្ឍន៍ប្រព័ន្ធដោយអនុលោមតាមតម្រូវការ និងស្តង់ដារគុណភាពល្អបំផុត។
    • ធ្វើការទំនាក់ទំនងជាប្រចាំជាមួយអតិថិជនអំពីវឌ្ឍនភាពការងារ។
    • រក្សាការសម្ងាត់រាល់ព័ត៌មានដែលទទួលបានពីអតិថិជន។
    • ផ្ដល់ការគាំទ្របច្ចេកទេសរយៈពេល [X] ខែបន្ទាប់ពីការដាក់ឱ្យដំណើរការ សម្រាប់ការកែតម្រូវ Bug ឬកំហុសឆ្គងដែលកើតចេញពីការអភិវឌ្ឍន៍។

    IV.២. ការទទួលខុសត្រូវរបស់អតិថិជន:

    • ផ្ដល់ព័ត៌មាន និងទិន្នន័យចាំបាច់ទាំងអស់ឱ្យបានទាន់ពេលវេលា។
    • ចូលរួមក្នុងការប្រជុំ និងផ្ដល់មតិកែលម្អលើការអភិវឌ្ឍន៍។
    • អនុវត្តការទូទាត់តាមកាលកំណត់ដែលបានព្រមព្រៀង។
    • ផ្ដល់បុគ្គលិកដើម្បីចូលរួមក្នុងការបណ្ដុះបណ្ដាល និងការធ្វើតេស្ត។

  5. V. កម្មសិទ្ធិបញ្ញា (Intellectual Property)

    បន្ទាប់ពីការទូទាត់ពេញលេញនូវតម្លៃសេវាកម្ម ប្រព័ន្ធកម្មវិធីដែលបានអភិវឌ្ឍន៍ទាំងអស់ រួមទាំងកូដប្រភព (Source Code) ឯកសាររចនា និងទ្រព្យសម្បត្តិបញ្ញាផ្សេងៗដែលកើតចេញពីគម្រោងនេះ នឹងក្លាយជាកម្មសិទ្ធិរបស់ **អតិថិជន** ទាំងស្រុង។

  6. VI. ការសម្ងាត់ (Confidentiality)

    ភាគីទាំងពីរយល់ព្រមរក្សាការសម្ងាត់រាល់ព័ត៌មានដែលទទួលបានពីភាគីម្ខាងទៀតក្នុងអំឡុងពេល និងក្រោយពេលបញ្ចប់កិច្ចសន្យានេះ។

  7. VII. ការកែប្រែវិសាលភាពការងារ (Change Management)

    រាល់ការផ្លាស់ប្តូរ ឬបន្ថែមមុខងារទៅលើវិសាលភាពការងារដែលបានព្រមព្រៀង នឹងត្រូវធ្វើឡើងជាលាយលក្ខណ៍អក្សរ និងទទួលបានការយល់ព្រមពីភាគីទាំងពីរ។ ការផ្លាស់ប្តូរទាំងនោះអាចប៉ះពាល់ដល់ពេលវេលា និងតម្លៃគម្រោង។

  8. VIII. ការបញ្ចប់កិច្ចសន្យា (Termination)

    ភាគីណាមួយអាចបញ្ចប់កិច្ចសន្យានេះដោយផ្ដល់ដំណឹងជាលាយលក្ខណ៍អក្សរ [X] ថ្ងៃជាមុន ប្រសិនបើភាគីម្ខាងទៀតមិនបានគោរពតាមលក្ខខណ្ឌនៃកិច្ចសន្យា។ ក្នុងករណីបញ្ចប់កិច្ចសន្យា ការទូទាត់នឹងត្រូវធ្វើឡើងសម្រាប់តែការងារដែលបានសម្រេច។

  9. IX. ច្បាប់អនុវត្ត និងយុត្តាធិការ (Governing Law and Jurisdiction)

    កិច្ចសន្យានេះត្រូវបានអនុវត្តតាមច្បាប់នៃព្រះរាជាណាចក្រកម្ពុជា។ រាល់វិវាទដែលកើតឡើងពីកិច្ចសន្យានេះ នឹងត្រូវដោះស្រាយដោយការចរចាប្រកបដោយសុឆន្ទៈ។ ប្រសិនបើការចរចាមិនអាចដោះស្រាយបាន វិវាទនឹងត្រូវបញ្ជូនទៅតុលាការដែលមានសមត្ថកិច្ចនៅព្រះរាជាណាចក្រកម្ពុជា។

ជាសក្ខីភាព ភាគីទាំងពីរបានចុះហត្ថលេខាលើកិច្ចសន្យានេះនៅថ្ងៃខែឆ្នាំដូចបានរៀបរាប់ខាងលើ។
